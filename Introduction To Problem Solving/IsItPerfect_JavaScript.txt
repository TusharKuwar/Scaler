const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
}); 
var count =0;
   rl.on('line', (A) => {
        if(count!=0)
   {
    var sum=0;
      for(var i=1;i<=A/2;i++){
        if(A%i == 0){
            sum = sum + i;
        }
   }
  
  if(A == sum){
      console.log("YES");     
   }
   else{
      console.log("NO");     
   }
   }
   count++;
});
